Cute animals for your console!


Usage:

```js
var Log = require('log.pets');

//log an cat in cmd
Log.cat()
//    /\_/\
//    >^.^<.---.
//   _'-`-'     )\
//  (6--\ |--\ (`.`-.
//      --'  --'  ``-'

//log an dog in cmd
Log.dog()
//           __
//  \ ______/ V`-,
//   }        /~~
//  /_)^ --,r'
// |b      |b

//log an penguin in cmd
Log.penguin()
// (o<
// //\
// V_/_ 

//log an turtle in cmd
Log.turtle()
//    ____
// __/-^-^\
//(' )^-^-^\)
// `^UU^^UU^

//log an dinosaur in cmd
Log.dinosaur()
//                __
//               / _)
//      _/\/\/\_/ /
//    _|         /
//  _|  (  | (  |
// /__.-'|_|--|_|

//log an cow in cmd
Log.cow()
//          __n__n__
//   .------`-\00/-'
//  /  ##  ## (oo)
// / \## __   ./
//    |//YY \|/
//    |||   |||

//log an bunny in cmd
Log.bunny()
//            /\ /|
//           |||| |
//            \ | \
//        _ _ /  @ @
//      /    \   =>X<=
//    /|      |   /
//    \|     /__| |
//      \_____\ \__\

//log an giraffe in cmd
Log.giraffe()
//     /)/)
//    ( ..\     
//    /'-._)   
//   /#/       
//  /#/

//log an unicorn in cmd
Log.unicorn()
//                                /
//                     __       //
//                     -\= \=\ //
//                   --=_\=---//=--
//                 -_==/  \/ //\/--
//                  ==/   /O   O\==--
//     _ _ _ _     /_/    \  ]  /--
//    /\ ( (- \    /       ] ] ]==-
//   (\ _\_\_\-\__/     \  (,_,)--
//  (\_/                 \     \-
//  \/      /       (   ( \  ] /)
//  /      (         \   \_ \./ )
//  (       \         \      )  \
//  (       /\_ _ _ _ /---/ /\_  \
//   \     / \     / ____/ /   \  \
//    (   /   )   / /  /__ )   (  )
//    (  )   / __/ '---`       / /
//    \  /   \ \             _/ /
//    ] ]     )_\_         /__\/
//    /_\     ]___\
//   (___)



//print nyan cat animation in cmd
// ATTENTION NYAN CAT STILL IN BETA... FOR EXIT FROM CMD USE Ctrl^C
Log.nyan();
// **********,------,
// #########v|   /\_/\
// ==========|__( ^ W^)
//             "  "
```
